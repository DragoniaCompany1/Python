üè† Analogi Rumah dan Alamat

Bayangkan variabel seperti rumah di kompleks perumahan: 

# Ini seperti membangun rumah dengan alamat "nama"
nama = "Budi Santoso"

# Alamat: nama
# Isi rumah: "Budi Santoso"
# Tipe rumah: String (teks)

üè† Komponen Variabel:

  • Nama variabel = Alamat rumah (nama)
  • Nilai variabel = Isi rumah ("Budi Santoso")
  • Tipe data = Jenis rumah (rumah teks, rumah angka, dll)

üîç Cara Cek Tipe Data 

# Function type() untuk cek tipe data

# Output: Tipe data dari Python adalah str

üîß String Operations yang Wajib Dikuasai

1. Concatenation (Penggabungan)

# Cara 1: Operator +
nama_depan = "John"
nama_belakang = "Doe"
nama_lengkap = nama_depan + " " + nama_belakang
print(nama_lengkap)  # John Doe

# Cara 2: F-strings (paling recommended!)
umur = 25
pesan = f"Halo, nama saya {nama_lengkap} dan saya berumur {umur} tahun"
print(pesan)  # Halo, nama saya John Doe dan saya berumur 25 tahun

# Cara 3: format() method
template = "Nama: {}, Umur: {}"
result = template.format(nama_lengkap, umur)
print(result)  # Nama: John Doe, Umur: 25

2. String Methods yang Super Berguna

teks = "  Python Programming adalah AWESOME!  "

# Cleaning methods
print(teks.strip())           # Hapus spasi di awal/akhir
print(teks.lower())           # Semua huruf kecil
print(teks.upper())           # Semua huruf besar
print(teks.title())           # Title Case

# Checking methods
print(teks.startswith("Python"))  # True/False
print(teks.endswith("!"))         # True/False
print("123".isdigit())            # True (semua angka)
print("abc".isalpha())            # True (semua huruf)

# Manipulation methods
print(teks.replace("AWESOME", "AMAZING"))  # Ganti kata
print(teks.split())                        # Split jadi list
print("-".join(["Python", "is", "cool"])) # Join dari list

3. String Indexing & Slicing

kata = "PYTHON"
#       012345  (index positif)
#      -654321  (index negatif)

# Indexing (ambil 1 karakter)
print(kata[0])   # P (karakter pertama)
print(kata[-1])  # N (karakter terakhir)
print(kata[2])   # T (karakter ketiga)

# Slicing (ambil beberapa karakter)
print(kata[0:3])   # PYT (dari index 0 sampai 2)
print(kata[2:])    # THON (dari index 2 sampai akhir)
print(kata[:4])    # PYTH (dari awal sampai index 3)
print(kata[::2])   # PTO (setiap 2 karakter)
print(kata[::-1])  # NOHTYP (reverse string)

üî¢ Numbers: Integer vs Float

üéØ Integer (int) - Angka Bulat 

# Berbagai cara menulis integer
umur = 25
jumlah_siswa = 30
tahun = 2024

# Integer bisa sangat besar di Python (unlimited precision!)
angka_besar = 123456789012345678901234567890
print(angka_besar)  # Python bisa handle!

# Berbagai base number system

üéØ Float - Angka Decimal 

# Berbagai cara menulis float
tinggi = 175.5
berat = 68.5
pi = 3.14159
suhu = -5.2

# Scientific notation
kecepatan_cahaya = 3e8      # 3 x 10^8 = 300,000,000
massa_elektron = 9.11e-31   # 9.11 x 10^-31

print(f"Kecepatan cahaya: {kecepatan_cahaya}")
print(f"Massa elektron: {massa_elektron}")

# Float precision issues (hati-hati!)
print(0.1 + 0.2)  # 0.30000000000000004 (bukan 0.3!)
print(0.1 + 0.2 == 0.3)  # False!

# Solusi: gunakan round()
print(round(0.1 + 0.2, 1))  # 0.3

‚úÖ Boolean: True atau False

Boolean adalah tipe data yang hanya punya 2 nilai: True atau False.

üéØ Boolean Basics 

# Boolean literals
is_student = True
is_graduated = False
has_job = True

print(f"Status mahasiswa: {is_student}")
print(f"Sudah lulus: {is_graduated}")
print(f"Punya kerja: {has_job}")

üîÑ Boolean dari Operasi Lain 

# Dari comparison
umur = 17
is_adult = umur >= 18
print(f"Sudah dewasa: {is_adult}")  # False

# Dari membership testing
buah_favorit = "apel"
is_citrus = buah_favorit in ["jeruk", "lemon", "limau"]
print(f"Buah citrus: {is_citrus}")  # False

# Dari string methods
email = "user@gmail.com"
is_gmail = email.endswith("@gmail.com")
print(f"Email Gmail: {is_gmail}")  # True

‚ö° Logical Operations 

# Operator AND, OR, NOT
is_weekend = True
is_holiday = False
have_money = True

# AND: semua harus True
can_travel = is_weekend and have_money
print(f"Bisa traveling: {can_travel}")  # True

# OR: salah satu True sudah cukup
is_free_day = is_weekend or is_holiday
print(f"Hari libur: {is_free_day}")  # True

# NOT: kebalikan
is_workday = not is_weekend
print(f"Hari kerja: {is_workday}")  # False

‚ö†Ô∏è Error Handling dalam Konversi 

# Safe conversion function
def safe_int_convert(value, default=0):
    try:
        return int(value)
    except (ValueError, TypeError):
        return default

print(safe_int_convert("123"))    # 123
print(safe_int_convert("hello"))  # 0 (default)
print(safe_int_convert(None))     # 0 (default)

üéÆ Project: Password Generator Sederhana 

import random
import string

def generate_password():
    print("üîê PASSWORD GENERATOR")
    print("=" * 25)
    
    # Input panjang password
    while True:
        try:
            length = int(input("Panjang password (4-20): "))
            if 4 <= length <= 20:
                break
            else:
                print("‚ö†Ô∏è Panjang password harus 4-20 karakter!")
        except ValueError:
            print("‚ùå Masukkan angka yang valid!")
    
    # Pilihan karakter
    use_letters = input("Gunakan huruf? (y/n): ").lower() == 'y'
    use_numbers = input("Gunakan angka? (y/n): ").lower() == 'y'
    use_symbols = input("Gunakan simbol? (y/n): ").lower() == 'y'
    
    # Build character set
    chars = ""
    if use_letters:
        chars += string.ascii_letters
    if use_numbers:
        chars += string.digits
    if use_symbols:
        chars += "!@#$%^&*"
    
    if not chars:
        print("‚ùå Minimal pilih satu jenis karakter!")
        return
    
    # Generate password
    password = ''.join(random.choice(chars) for _ in range(length))
    
    print(f"\nüéØ Password Anda: {password}")
    print(f"üìè Panjang: {len(password)} karakter")

üéØ Project: Text Analyzer Mini 

def analyze_text():
    print("üìù TEXT ANALYZER")
    print("=" * 20)
    
    text = input("Masukkan teks: ")
    
    if not text.strip():
        print("‚ùå Teks tidak boleh kosong!")
        return
    
    # Analisis dasar
    char_count = len(text)
    char_no_space = len(text.replace(" ", ""))
    word_count = len(text.split())
    sentence_count = text.count('.') + text.count('!') + text.count('?')
    
    # Analisis karakter
    uppercase = sum(1 for c in text if c.isupper())
    lowercase = sum(1 for c in text if c.islower())
    digits = sum(1 for c in text if c.isdigit())
    spaces = text.count(' ')
    
    # Output hasil
    print(f"\nüìä HASIL ANALISIS:")
    print(f"üìè Total karakter: {char_count}")
    print(f"üî§ Karakter (tanpa spasi): {char_no_space}")
    print(f"üìù Jumlah kata: {word_count}")
    print(f"üìÑ Jumlah kalimat: {sentence_count}")
    print(f"üî† Huruf besar: {uppercase}")
    print(f"üî° Huruf kecil: {lowercase}")
    print(f"üî¢ Angka: {digits}")
    print(f"‚≠ê Spasi: {spaces}")
    
    # Kata terpanjang
    words = text.split()
    if words:
        longest_word = max(words, key=len)
        print(f"üìè Kata terpanjang: '{longest_word}' ({len(longest_word)} huruf)")
    
    # Reading time estimate (rata-rata 200 kata per menit)
    reading_time = max(1, word_count // 200)
    print(f"‚è±Ô∏è Estimasi waktu baca: {reading_time} menit")

# Buat function untuk cek apakah string adalah palindrom
# Contoh: "racecar", "madam", "level"

def is_palindrome(text):
    # Hilangkan spasi dan ubah ke huruf kecil
    clean_text = ''.join(c.lower() for c in text if c.isalnum())
    return clean_text == clean_text[::-1]

# Test cases
print(is_palindrome("racecar"))  # True
print(is_palindrome("hello"))    # False
print(is_palindrome("A man a plan a canal Panama"))  # True (ignore spaces & case)

import random

def guessing_game():
    # Buat game tebak angka 1-100
    # User punya 7 kesempatan
    # Berikan hint "terlalu besar" atau "terlalu kecil"
    
    number = random.randint(1, 100)
    attempts = 7
    
    print("üéÆ GAME TEBAK ANGKA")
    print("Saya memikirkan angka 1-100")
    print(f"Anda punya {attempts} kesempatan")
    
    while attempts > 0:
        try:
            guess = int(input("Tebak angka: "))
        except ValueError:
            print("‚ùå Masukkan angka yang valid!")
            continue
        
        if guess == number:
            print("üéâ Selamat! Tebakan Anda benar!")
            return
        elif guess > number:
            print("‚¨áÔ∏è Terlalu besar!")
        else:
            print("‚¨ÜÔ∏è Terlalu kecil!")
        
        attempts -= 1
        print(f"Sisa kesempatan: {attempts}")
    
    print(f"üòû Kesempatan habis! Angka yang benar adalah {number}")
